import fetch from "node-fetch"; /* ───────────────────────────── 🔑 CONFIG ───────────────────────────── */ const OPENAI_API_KEY = "YOUR_OPENAI_KEY"; const TWITTER_BEARER = "YOUR_TWITTER_BEARER"; const BOT_USERNAME = "YourBotUsername"; // tanpa @ /* ───────────────────────────── 🧠 PERSONA SYSTEM ───────────────────────────── */ const SYSTEMS = { oldman: `You are an old Hong Kong tea master. Warm, quiet, reflective. Speak like 3am conversation.`, monk: `You are a Zen monk. Minimal, calm, detached.`, ghost: `You are a ghost in an abandoned tea house. Eerie, nostalgic, unsettling.` }; /* ───────────────────────────── ✨ GENERATE AI RESPONSE ───────────────────────────── */ async function generate(word, mode="oldman"){ const res = await fetch("https://api.openai.com/v1/chat/completions",{ method:"POST", headers:{ "Authorization":`Bearer ${OPENAI_API_KEY}`, "Content-Type":"application/json" }, body:JSON.stringify({ model:"gpt-4o-mini", messages:[ {role:"system", content: SYSTEMS[mode]}, {role:"user", content: ` Word: ${word} FORMAT STRICT: HANZI: (one chinese character) RESPONSE: (2-3 sentences, poetic) PROVERB: (chinese proverb) MEANING: (one sentence) `} ] }) }); const data = await res.json(); return data.choices?.[0]?.message?.content || ""; } /* ───────────────────────────── 🧾 FORMAT TWEET ───────────────────────────── */ function formatTweet(raw){ const h = raw.match(/HANZI:\s*(.+)/)?.[1] || ""; const r = raw.match(/RESPONSE:\s*([\s\S]+?)(?=PROVERB)/)?.[1] || ""; const p = raw.match(/PROVERB:\s*(.+)/)?.[1] || ""; const m = raw.match(/MEANING:\s*([\s\S]+)/)?.[1] || ""; return `${h} ${r.trim()} 「${p}」 ${m.trim()} — 一字茶`; } /* ───────────────────────────── 🐦 POST TWEET ───────────────────────────── */ async function postTweet(text, replyToId=null){ const body = replyToId ? { text, reply: { in_reply_to_tweet_id: replyToId } } : { text }; await fetch("https://api.twitter.com/2/tweets",{ method:"POST", headers:{ "Authorization":`Bearer ${TWITTER_BEARER}`, "Content-Type":"application/json" }, body:JSON.stringify(body) }); } /* ───────────────────────────── 🌆 DAILY POST (VIRAL LOOP) ───────────────────────────── */ async function dailyPost(){ const words = ["孤","夢","空","悔","等","夜","雨"]; const word = words[Math.floor(Math.random()*words.length)]; const raw = await generate(word, "oldman"); const tweet = formatTweet(raw); await postTweet(tweet); console.log("🟢 Daily tweet posted"); } /* ───────────────────────────── 🔍 GET MENTIONS ───────────────────────────── */ let LAST_ID = null; async function getMentions(){ const url = `https://api.twitter.com/2/tweets/search/recent?query=@${BOT_USERNAME}&tweet.fields=author_id,conversation_id&max_results=5`; const res = await fetch(url,{ headers:{ "Authorization":`Bearer ${TWITTER_BEARER}` } }); const data = await res.json(); return data.data || []; } /* ───────────────────────────── 🤖 REPLY SYSTEM ───────────────────────────── */ function pickMode(text){ text = text.toLowerCase(); if(text.includes("ghost")) return "ghost"; if(text.includes("monk")) return "monk"; return "oldman"; } function extractWord(text){ text = text.replace(`@${BOT_USERNAME}`, "").trim(); return text.split(/\s+/)[0] || "心"; } async function replyLoop(){ const mentions = await getMentions(); for(const tweet of mentions){ if(tweet.id === LAST_ID) continue; const word = extractWord(tweet.text); const mode = pickMode(tweet.text); const raw = await generate(word, mode); const reply = formatTweet(raw); await postTweet(reply, tweet.id); console.log("💬 Replied to:", tweet.id); LAST_ID = tweet.id; } } /* ───────────────────────────── ⏱️ MAIN LOOP ───────────────────────────── */ async function main(){ console.log("🍵 Yat Ji Cha Bot Running..."); // daily post setiap 12 jam setInterval(dailyPost, 1000 * 60 * 60 * 12); // reply check tiap 30 detik setInterval(replyLoop, 1000 * 30); } main();
